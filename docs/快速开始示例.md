# 快速开始：提取第一个共享组件

## 示例：提取导航栏组件

这个示例展示如何把现有的导航栏提取成共享组件。

### 步骤1：创建共享组件目录

```bash
mkdir -p shared/components/navbar
```

### 步骤2：提取导航栏HTML结构

**原来的代码**（在 `index.html` 中）：
```html
<nav class="navbar navbar-expand-lg navbar-light bg-light">
    <div class="container-fluid">
        <a class="navbar-brand" href="#home">
            <i class="bi bi-pencil-square"></i> 看拼音写词
        </a>
        <ul class="navbar-nav ms-auto">
            <li class="nav-item">
                <a class="nav-link" href="#home">首页</a>
            </li>
            <li class="nav-item">
                <a class="nav-link" href="#errorbook">错题本</a>
            </li>
        </ul>
    </div>
</nav>
```

**提取后的组件**（`shared/components/navbar/navbar.js`）：
```javascript
/**
 * 共享导航栏组件
 */
class Navbar {
    constructor(config = {}) {
        this.appName = config.appName || '应用名称';
        this.appIcon = config.appIcon || 'bi-app';
        this.menuItems = config.menuItems || [
            { label: '首页', href: '#home' },
            { label: '错题本', href: '#errorbook' }
        ];
        this.showAppSwitcher = config.showAppSwitcher || false;
    }
    
    /**
     * 渲染导航栏
     * @param {string} containerId - 容器ID
     */
    render(containerId) {
        const container = document.getElementById(containerId);
        if (!container) {
            console.error(`容器 ${containerId} 不存在`);
            return;
        }
        
        container.innerHTML = this.getHTML();
        this.bindEvents();
    }
    
    /**
     * 生成HTML
     */
    getHTML() {
        const appSwitcherHTML = this.showAppSwitcher 
            ? '<button class="btn btn-sm btn-outline-secondary" id="app-switcher-btn"><i class="bi bi-grid"></i> 应用</button>'
            : '';
        
        const menuItemsHTML = this.menuItems.map(item => 
            `<li class="nav-item">
                <a class="nav-link" href="${item.href}">${item.label}</a>
            </li>`
        ).join('');
        
        return `
            <nav class="navbar navbar-expand-lg navbar-light bg-light">
                <div class="container-fluid">
                    <a class="navbar-brand" href="#home">
                        <i class="bi ${this.appIcon}"></i> ${this.appName}
                    </a>
                    <button class="navbar-toggler" type="button" data-bs-toggle="collapse" data-bs-target="#navbarNav">
                        <span class="navbar-toggler-icon"></span>
                    </button>
                    <div class="collapse navbar-collapse" id="navbarNav">
                        <ul class="navbar-nav ms-auto">
                            ${menuItemsHTML}
                            ${appSwitcherHTML ? `<li class="nav-item ms-2">${appSwitcherHTML}</li>` : ''}
                            <li class="nav-item ms-2">
                                <button class="btn btn-sm btn-outline-secondary" id="theme-toggle-btn">
                                    <i class="bi bi-brightness-high"></i> 主题
                                </button>
                            </li>
                        </ul>
                    </div>
                </div>
            </nav>
        `;
    }
    
    /**
     * 绑定事件
     */
    bindEvents() {
        // 主题切换
        const themeBtn = document.getElementById('theme-toggle-btn');
        if (themeBtn) {
            themeBtn.addEventListener('click', () => {
                this.toggleTheme();
            });
        }
        
        // 应用切换器
        if (this.showAppSwitcher) {
            const switcherBtn = document.getElementById('app-switcher-btn');
            if (switcherBtn) {
                switcherBtn.addEventListener('click', () => {
                    this.showAppSwitcher();
                });
            }
        }
    }
    
    /**
     * 切换主题
     */
    toggleTheme() {
        const currentTheme = document.documentElement.getAttribute('data-theme');
        const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
        document.documentElement.setAttribute('data-theme', newTheme);
        localStorage.setItem('theme', newTheme);
    }
}

// 导出组件
if (typeof module !== 'undefined' && module.exports) {
    module.exports = Navbar;
} else {
    window.Navbar = Navbar;
}
```

### 步骤3：创建组件样式（如果需要）

`shared/components/navbar/navbar.css`：
```css
/* 导航栏组件样式 */
.navbar-brand {
    font-weight: 600;
}

/* 应用切换器按钮 */
#app-switcher-btn {
    margin-left: 0.5rem;
}
```

### 步骤4：在应用中使用

**修改 `index.html`**：
```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>看拼音写词</title>
    <!-- 引入共享组件样式 -->
    <link rel="stylesheet" href="shared/components/navbar/navbar.css">
    <!-- 其他样式 -->
</head>
<body>
    <!-- 导航栏容器 -->
    <div id="navbar-container"></div>
    
    <!-- 其他内容 -->
    <div id="app-content">
        <!-- ... -->
    </div>
    
    <!-- 引入共享组件 -->
    <script src="shared/components/navbar/navbar.js"></script>
    <!-- 应用主逻辑 -->
    <script>
        // 初始化导航栏
        const navbar = new Navbar({
            appName: '看拼音写词',
            appIcon: 'bi-pencil-square',
            menuItems: [
                { label: '首页', href: '#home' },
                { label: '错题本', href: '#errorbook' },
                { label: '题库管理', href: '#wordbank' }
            ],
            showAppSwitcher: true  // 如果是在应用中心，显示应用切换器
        });
        
        navbar.render('navbar-container');
        
        // 其他应用逻辑...
    </script>
</body>
</html>
```

### 步骤5：在第二个应用中使用

**`apps/math-practice/index.html`**：
```html
<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>加减法练习</title>
    <link rel="stylesheet" href="../../shared/components/navbar/navbar.css">
</head>
<body>
    <div id="navbar-container"></div>
    
    <script src="../../shared/components/navbar/navbar.js"></script>
    <script>
        // 使用相同的导航栏组件，但配置不同
        const navbar = new Navbar({
            appName: '加减法练习',
            appIcon: 'bi-calculator',
            menuItems: [
                { label: '首页', href: '#home' },
                { label: '错题本', href: '#errorbook' }
            ],
            showAppSwitcher: true
        });
        
        navbar.render('navbar-container');
    </script>
</body>
</html>
```

## 效果

- ✅ 导航栏代码只写一次
- ✅ 两个应用共用同一个组件
- ✅ 修改导航栏时，两个应用都更新
- ✅ 每个应用可以自定义配置（名称、图标、菜单项）
